<div class="app">
<HelloWorld msg="hi" />
</div>
</template>

<script setup>
import HelloWorld from './HelloWorld.vue'
</script>

<style scoped>
.app { color: red; }
</style>

<template>
  <div class="app">
    <HelloWorld msg="hi" />
  </div>
</template>

import HelloWorld from './HelloWorld.vue'
</script>

<style scoped>

== svelte expressions stay verbatim ==
<script>
  let count = 0;
</script>

<button on:click={() => count += 1}>
clicks: {count}
</button>

[expect]
<script>
  let count = 0;
</script>

<button on:click={() => count += 1}>
clicks: {count}
</button>
